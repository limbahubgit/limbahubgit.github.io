<script>
(function() {
  // Generate film holes dynamically
  function makeHoles(containerId) {
    const el = document.getElementById(containerId);
    for (let i = 0; i < 30; i++) {
      const hole = document.createElement('div');
      hole.className = 'film-hole';
      el.appendChild(hole);
    }
  }
  makeHoles('topHoles');
  makeHoles('bottomHoles');

  // Scroll with mouse wheel horizontally
  const track = document.getElementById('filmTrack');
  const outer = document.getElementById('filmOuter');

  outer.addEventListener('wheel', function(e) {
    e.preventDefault();
    track.scrollLeft += e.deltaY * 1.5;
  }, { passive: false });

  // Drag to scroll
  let isDown = false;
  let startX;
  let scrollLeft;

  outer.addEventListener('mousedown', (e) => {
    isDown = true;
    startX = e.pageX - outer.offsetLeft;
    scrollLeft = track.scrollLeft;
  });
  outer.addEventListener('mouseleave', () => isDown = false);
  outer.addEventListener('mouseup', () => isDown = false);
  outer.addEventListener('mousemove', (e) => {
    if (!isDown) return;
    e.preventDefault();
    const x = e.pageX - outer.offsetLeft;
    const walk = (x - startX) * 2;
    track.scrollLeft = scrollLeft - walk;
  });

  // Touch support
  let touchStartX = 0;
  let touchScrollLeft = 0;
  outer.addEventListener('touchstart', (e) => {
    touchStartX = e.touches[0].pageX;
    touchScrollLeft = track.scrollLeft;
  });
  outer.addEventListener('touchmove', (e) => {
    const x = e.touches[0].pageX;
    const walk = (touchStartX - x) * 1.5;
    track.scrollLeft = touchScrollLeft + walk;
  });
})();
</script>
